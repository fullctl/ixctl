
### Installtion and setup

1. setup database user and db in mysql
    create database ${USER}_django_ixctl;
    grant all on ${USER}_django_ixctl.* to '${USER}_django_ixctl'@'localhost' identified by '$(uuidgen)';
    flush privileges;

2. cp ixctl/settings.d/01-local.conf.dist ixctl/settings.d/01-local.conf
3. edit ixctl/settings.d/01-local.conf and setup db
4. pipenv install --dev
5. pipenv shell
6. cd src 
7. python manage.py migrate
8. python manage.py runserver

### cron job to regen routeserver config

* * * * * python manage.py ixctl_rsconf_generate
